```bash
# opis
$ komenda
# wypisanie branchy
Git branch lub git br 

#stworzenie brancha 
Git branch nazwa_brancha lub git br nazwa_brancha

# przejście na brancha 
Git checkout nazwa_brancha lub git co nazwa_brancha

# sprawdzenie statusów plików 
Git status lub git st

#dodanie plików 
Git add --all lub Git add .

#skomitowanie plików
Git commit -am 'opis zmian' lub git co -am 'opis zmian'

#wysłanie brancha do mastera
Git push origin nazwa_brancha 

#pobranie najnowszej wersji brancha
Git pull

#usuwanie brancha
Git branch -d nazwa_brancha

```
